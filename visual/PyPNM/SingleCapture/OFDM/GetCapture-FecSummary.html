// Visualization Script const template = `

FEC Summary - Codeword Analysis

{{#each channels}}

Channel ID: {{channel_id}}

{{#each profiles}}

Profile {{profile}}

{{/each}}
{{/each}}
`; function constructVisualizerPayload() { const responseData = pm.response.json(); const channels = []; if (responseData.data && responseData.data.analysis) { responseData.data.analysis.forEach(function(channelData) { const channelObj = { channel_id: channelData.channel_id, profiles: [] }; channelData.profiles.forEach(function(profileData) { const timestampLabels = profileData.codewords.timestamps.map(function(ts) { return new Date(ts * 1000).toLocaleTimeString(); }); channelObj.profiles.push({ profile: profileData.profile, timestampLabels: timestampLabels, total_codewords: profileData.codewords.total_codewords, corrected: profileData.codewords.corrected, uncorrected: profileData.codewords.uncorrected }); }); channels.push(channelObj); }); } return { channels: channels }; } pm.visualizer.set(template, constructVisualizerPayload());
