`; function sanitizeMac(value, fallback) { var fb = (fallback === undefined || fallback === null) ? 'N/A' : fallback; if (value === undefined || value === null) return fb; var text = String(value).trim(); if (!text) return fb; var compact = text.replace(/[^0-9a-f]/gi, '').toLowerCase(); if (compact.length !== 12) return text; return compact.match(/.{1,2}/g).join(':'); } function sanitizeSystemDescription(_sys) { var sys = (_sys && typeof _sys === 'object') ? _sys : {}; function present(v) { return v !== undefined && v !== null && String(v).trim() !== ''; } return { HW_REV: present(sys.HW_REV) ? String(sys.HW_REV).trim() : 'N/A', VENDOR: present(sys.VENDOR) ? String(sys.VENDOR).trim() : 'N/A', BOOTR: present(sys.BOOTR) ? String(sys.BOOTR).trim() : 'N/A', SW_REV: present(sys.SW_REV) ? String(sys.SW_REV).trim() : 'N/A', MODEL: present(sys.MODEL) ? String(sys.MODEL).trim() : 'N/A' }; } function mergeDeviceInfo(base, incoming) { if (!base) return incoming; if (!incoming) return base; if (base.macAddress === 'N/A' && incoming.macAddress !== 'N/A') base.macAddress = incoming.macAddress; ['MODEL','VENDOR','SW_REV','HW_REV','BOOTR'].forEach(function (k) { if (base[k] === 'N/A' && incoming[k] !== 'N/A') base[k] = incoming[k]; }); return base; } function formatCaptureTime(raw) { if (raw === undefined || raw === null || raw === '') return null; if (typeof raw === 'number' && isFinite(raw)) { var ms = raw > 1e12 ? raw : raw * 1000; var d = new Date(ms); if (isNaN(d.getTime())) return null; return d.toISOString().slice(0, 19).replace('T', ' ') + ' UTC'; } var n = Number(raw); if (!isNaN(n) && isFinite(n)) return formatCaptureTime(n); var d2 = new Date(raw); if (isNaN(d2.getTime())) return String(raw); return d2.toISOString().slice(0, 19).replace('T', ' ') + ' UTC'; } function constructVisualizerPayload() { var response = pm.response.json(); var analyses = (response && response.data && Array.isArray(response.data.analyses)) ? response.data.analyses : []; var measurementStats = (response && response.data && Array.isArray(response.data.measurement_stats)) ? response.data.measurement_stats : []; function safeNum(v, fallback) { return (typeof v === 'number' && isFinite(v)) ? v : fallback; } function toMHz(vHz) { return (typeof vHz === 'number' && isFinite(vHz)) ? (vHz / 1e6).toFixed(0) : 'N/A'; } function toMHzFixed(vHz, p) { return (typeof vHz === 'number' && isFinite(vHz)) ? (vHz / 1e6).toFixed(p) : 'N/A'; } function measurementKey(firstHz, lastHz, segSpanHz) { return [firstHz, lastHz, segSpanHz].map(function (v) { return (typeof v === 'number' && isFinite(v)) ? String(v) : 'na'; }).join('|'); } function computeStats(mags) { if (!Array.isArray(mags) || mags.length === 0) return { avg: 'N/A', min: 'N/A', max: 'N/A' }; var sum = 0, min = Infinity, max = -Infinity, n = 0; for (var i = 0; i < mags.length; i++) { var x = mags[i]; if (typeof x !== 'number' || !isFinite(x)) continue; sum += x; n += 1; if (x < min) min = x; if (x > max) max = x; } var avg = (n > 0) ? (sum / n) : NaN; return { avg: isFinite(avg) ? avg.toFixed(2) : 'N/A', min: isFinite(min) ? min.toFixed(2) : 'N/A', max: isFinite(max) ? max.toFixed(2) : 'N/A' }; } var deviceInfo = { macAddress: 'N/A', MODEL: 'LCPET-3', VENDOR: 'LANCity', SW_REV: '1.0.0', HW_REV: '1.0', BOOTR: 'NONE' }; var captureTime = null; var measurementByRange = {}; measurementStats.forEach(function (m) { var e = (m && m.entry) ? m.entry : {}; var cs = (((m || {}).channel_stats || {}).entry) || {}; var first = safeNum(e.docsIf3CmSpectrumAnalysisCtrlCmdFirstSegmentCenterFrequency, NaN); var last = safeNum(e.docsIf3CmSpectrumAnalysisCtrlCmdLastSegmentCenterFrequency, NaN); var seg = safeNum(e.docsIf3CmSpectrumAnalysisCtrlCmdSegmentFrequencySpan, NaN); var key = measurementKey(first, last, seg); if (!measurementByRange[key]) { measurementByRange[key] = { fileName: e.docsIf3CmSpectrumAnalysisCtrlCmdFileName || 'N/A', measStatus: e.docsIf3CmSpectrumAnalysisCtrlCmdMeasStatus || 'N/A', scqamChannelId: (typeof cs.docsIfDownChannelId === 'number') ? cs.docsIfDownChannelId : null, scqamFrequencyHz: safeNum(cs.docsIfDownChannelFrequency, NaN), scqamModulation: cs.docsIfDownChannelModulation || 'N/A' }; } }); var byChannel = {}; analyses.forEach(function (a) { if (!a) return; var sys = sanitizeSystemDescription((((a || {}).device_details || {}).system_description) || {}); deviceInfo = mergeDeviceInfo(deviceInfo, { macAddress: sanitizeMac(a.mac_address || response.mac_address, 'N/A'), MODEL: sys.MODEL, VENDOR: sys.VENDOR, SW_REV: sys.SW_REV, HW_REV: sys.HW_REV, BOOTR: sys.BOOTR }); if (!captureTime) captureTime = formatCaptureTime((((a || {}).pnm_header) || {}).capture_time); var cap = a.capture_parameters || {}; var sig = a.signal_analysis || {}; var firstHz = safeNum(cap.first_segment_center_freq, NaN); var lastHz = safeNum(cap.last_segment_center_freq, NaN); var segSpanHz = safeNum(cap.segment_freq_span, NaN); var centerHz = (isFinite(firstHz) && isFinite(lastHz)) ? (firstHz + lastHz) / 2 : NaN; var stats = computeStats(sig.magnitudes || []); var measurement = measurementByRange[measurementKey(firstHz, lastHz, segSpanHz)] || null; var chId = (measurement && typeof measurement.scqamChannelId === 'number') ? measurement.scqamChannelId : ((typeof a.channel_id === 'number') ? a.channel_id : 0); if (!byChannel[chId]) { byChannel[chId] = { channelId: chId, captures: [], tableRows: [], channelSummary: { freqRangeMHz: 'N/A', centerMHz: 'N/A', binBandwidthMHz: 'N/A', avgChannelPower: 'N/A', windowAveragePoints: 'N/A', modulation: 'N/A' } }; } var captureLabel = 'Capture ' + (byChannel[chId].captures.length + 1) + (isFinite(centerHz) ? (' @ ' + toMHz(centerHz) + ' MHz') : ''); byChannel[chId].captures.push({ captureLabel: captureLabel, capture_parameters: cap, signal_analysis: sig, measurement: measurement }); byChannel[chId].tableRows.push({ captureLabel: captureLabel, freqRangeMHz: (isFinite(firstHz) ? toMHz(firstHz) : 'N/A') + ' - ' + (isFinite(lastHz) ? toMHz(lastHz) : 'N/A'), centerMHz: isFinite(centerHz) ? toMHz(centerHz) : 'N/A', avgPower: stats.avg, minMaxPower: stats.min + ' / ' + stats.max, measurementStatus: measurement ? measurement.measStatus : 'N/A', measurementFileName: measurement ? measurement.fileName : 'N/A' }); if (byChannel[chId].channelSummary.freqRangeMHz === 'N/A') { byChannel[chId].channelSummary = { freqRangeMHz: (isFinite(firstHz) ? toMHz(firstHz) : 'N/A') + ' - ' + (isFinite(lastHz) ? toMHz(lastHz) : 'N/A'), centerMHz: isFinite(centerHz) ? toMHz(centerHz) : 'N/A', binBandwidthMHz: (sig.bin_bandwidth !== undefined && sig.bin_bandwidth !== null) ? toMHzFixed(sig.bin_bandwidth, 1) : 'N/A', avgChannelPower: (sig.channel_power_dbmv !== undefined && sig.channel_power_dbmv !== null && isFinite(Number(sig.channel_power_dbmv))) ? Number(sig.channel_power_dbmv).toFixed(2) : 'N/A', windowAveragePoints: (sig.window_average && sig.window_average.points !== undefined && sig.window_average.points !== null) ? String(sig.window_average.points) : 'N/A', modulation: (measurement && measurement.scqamModulation) ? String(measurement.scqamModulation) : 'N/A' }; } }); var channels = Object.keys(byChannel).map(function (k) { return byChannel[k]; }).sort(function (a, b) { return a.channelId - b.channelId; }).map(function (ch) { ch.captureCount = (ch.captures || []).length; return ch; }); var totalCaptures = 0; channels.forEach(function (ch) { totalCaptures += ch.captureCount || 0; }); return { deviceInfo: deviceInfo, captureTime: captureTime, channels: channels, channelCount: channels.length, totalCaptures: totalCaptures }; } pm.visualizer.set(template, constructVisualizerPayload());
