// Visualization Script for OFDM Modulation Profile Analysis const template = `

OFDM Modulation Profile Analysis - Shannon Min MER vs Frequency

{{#each channels}}
Channel ID: {{channelId}} | Device: {{deviceModel}}
{{/each}}
`; function sanitizeSystemDescription(_sys) { return { HW_REV: '1.0', VENDOR: 'LANCity', BOOTR: 'NONE', SW_REV: '1.0.0', MODEL: 'LCPET-3' }; } function constructVisualizerPayload() { var response = pm.response.json(); if (!response.data || !response.data.analysis) { return { channels: [] }; } var channels = response.data.analysis.map(function(channelData) { var deviceModel = sanitizeSystemDescription( channelData.device_details && channelData.device_details.system_description ? channelData.device_details.system_description : {} ).MODEL; var profiles = channelData.profiles.map(function(profile) { var frequencies = profile.carrier_values.frequency.map(function(freq) { return (freq / 1000000).toFixed(1); }); var modulation = 'Unknown'; if (profile.carrier_values.modulation && profile.carrier_values.modulation.length > 0) { modulation = profile.carrier_values.modulation[0].replace('qam_', 'QAM-'); } return { profileId: profile.profile_id, modulation: modulation, data: profile.carrier_values.shannon_min_mer }; }); var frequencies = []; if (channelData.profiles.length > 0 && channelData.profiles[0].carrier_values && channelData.profiles[0].carrier_values.frequency) { frequencies = channelData.profiles[0].carrier_values.frequency.map(function(freq) { return (freq / 1000000).toFixed(1); }); } return { channelId: channelData.channel_id, deviceModel: deviceModel, frequencies: frequencies, profiles: profiles }; }); return { channels: channels }; } pm.visualizer.set(template, constructVisualizerPayload());
