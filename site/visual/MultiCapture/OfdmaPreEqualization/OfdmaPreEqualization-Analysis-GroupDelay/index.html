function constructVisualizerPayload() {
    var response = pm.response.json();
    var results = response.data && response.data.results ? response.data.results : [];
    var hasData = results.length > 0;
    var analysisType = response.data && response.data.analysis_type ? response.data.analysis_type : 'UNKNOWN';

    var datasets = [];
    var masterFrequencies = [];
    var displayUnit = '';
    var yAxisLabel = '';
    var chartTitle = '';

    if (analysisType === 'GROUP_DELAY') {
        // Handle Group Delay analysis
        chartTitle = 'Group Delay vs Frequency';

        // Determine the best unit based on magnitude
        var allValues = [];
        results.forEach(function(result) {
            if (result.group_delay_us) {
                allValues = allValues.concat(result.group_delay_us);
            }
        });

        var maxAbsValue = 0;
        allValues.forEach(function(val) {
            var absVal = Math.abs(val);
            if (absVal > maxAbsValue) maxAbsValue = absVal;
        });

        // Choose unit: if max value is very small (< 0.001 us), use nanoseconds
        var multiplier = 1;
        if (maxAbsValue < 0.001 && maxAbsValue > 0) {
            displayUnit = 'ns';
            multiplier = 1000;
        } else {
            displayUnit = 'μs';
        }
        yAxisLabel = 'Group Delay (' + displayUnit + ')';

        results.forEach(function(result, index) {
            var channelLabel = result.channel_id !== undefined ? 
                'Channel ' + result.channel_id : 
                'Result ' + (index + 1);

            var groupDelayData = [];
            if (result.group_delay_us) {
                groupDelayData = result.group_delay_us.map(function(val) {
                    return val * multiplier;
                });
            }

            datasets.push({
                label: channelLabel,
                data: groupDelayData
            });

            if (result.frequency && result.frequency.length > masterFrequencies.length) {
                masterFrequencies = result.frequency;
            }
        });

    } else if (analysisType === 'MIN_AVG_MAX') {
        // Handle Min/Avg/Max analysis
        chartTitle = 'Min/Avg/Max vs Frequency';
        yAxisLabel = 'RxMER (dB)';

        results.forEach(function(result, index) {
            var channelLabel = result.channel_id !== undefined ? 
                'Ch ' + result.channel_id : 
                'Result ' + (index + 1);

            // Add Min, Avg, Max datasets for each channel
            if (result.min) {
                datasets.push({
                    label: channelLabel + ' Min',
                    data: result.min
                });
            }
            if (result.avg) {
                datasets.push({
                    label: channelLabel + ' Avg',
                    data: result.avg
                });
            }
            if (result.max) {
                datasets.push({
                    label: channelLabel + ' Max',
                    data: result.max
                });
            }

            if (result.frequency && result.frequency.length > masterFrequencies.length) {
                masterFrequencies = result.frequency;
            }
        });

    } else {
        // Unknown analysis type - try to plot whatever data is available
        chartTitle = analysisType + ' Analysis';
        yAxisLabel = 'Value';

        results.forEach(function(result, index) {
            var channelLabel = result.channel_id !== undefined ? 
                'Channel ' + result.channel_id : 
                'Result ' + (index + 1);

            // Try to find any array data to plot
            Object.keys(result).forEach(function(key) {
                if (Array.isArray(result[key]) && key !== 'frequency') {
                    datasets.push({
                        label: channelLabel + ' ' + key,
                        data: result[key]
                    });
                }
            });

            if (result.frequency && result.frequency.length > masterFrequencies.length) {
                masterFrequencies = result.frequency;
            }
        });
    }

    // Convert frequencies to MHz for display
    var frequencyLabels = masterFrequencies.map(function(freq) {
        return (freq / 1000000).toFixed(2);
    });

    return {
        mac_address: response.mac_address || 'N/A',
        status: response.status !== undefined ? response.status : 'N/A',
        message: response.message || 'N/A',
        analysis_type: analysisType,
        num_channels: results.length,
        display_unit: displayUnit,
        has_data: hasData,
        datasets: datasets,
        frequency_labels: frequencyLabels,
        y_axis_label: yAxisLabel,
        chart_title: chartTitle
    };
}
