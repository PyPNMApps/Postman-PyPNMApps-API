String(item.subcarrier_zero_frequency) : ''; } if ((!Array.isArray(channelRecord.systemDescriptionRows) || channelRecord.systemDescriptionRows.length === 0) && Array.isArray(systemDescriptionRows) && systemDescriptionRows.length) { channelRecord.systemDescriptionRows = systemDescriptionRows; } if (channelRecord.deviceInfo) { ['MODEL', 'VENDOR', 'SW_REV', 'HW_REV', 'BOOTR'].forEach(key => { if (channelRecord.deviceInfo[key] === 'N/A' && deviceInfo[key] !== 'N/A') { channelRecord.deviceInfo[key] = deviceInfo[key]; } }); } const ver = item.pnm_header && item.pnm_header.file_type_version !== undefined ? item.pnm_header.file_type_version : 'N/A'; const freqHz = item.carrier_values && Array.isArray(item.carrier_values.frequency) ? item.carrier_values.frequency : []; const frequencies = freqHz.map(f => { if (typeof f !== 'number' || !isFinite(f)) return 'N/A'; return (f / 1000000).toFixed(3); }); const magnitudes = item.carrier_values && Array.isArray(item.carrier_values.magnitudes) ? item.carrier_values.magnitudes : []; const groupDelays = item.carrier_values && item.carrier_values.group_delay && Array.isArray(item.carrier_values.group_delay.magnitude) ? item.carrier_values.group_delay.magnitude : []; const stats = item.signal_statistics || {}; const echoesRaw = item.echo && item.echo.report && Array.isArray(item.echo.report.echoes) ? item.echo.report.echoes : []; const echoes = echoesRaw.map(echo => ({ binIndex: echo.bin_index, time: (typeof echo.time_s === 'number' ?
