`; } const bins = Math.min(freq.length, N); const { minV, maxV } = matrixRange(values); const gridW = bins * CELL_PX; const gridH = T * CELL_PX; // Axis label spacing (sparse) const xEvery = Math.max(1, Math.floor(bins / MAX_X_LABELS)); const yEvery = Math.max(1, Math.floor(T / MAX_Y_LABELS)); // Build rects without stroke + without // Note: SVG with many <rect> can still be heavy. 4x4 helps (fewer rows/cols if you pre-aggregate), // but this removes the tooltip/title overhead and stroke. let rects = ""; for (let r = 0; r < T; r++) { const row = Array.isArray(values[r]) ? ts[r] : []; const t0 = tsRow.length > 0 ? tsRow[0] : null; yLabels += `<text x="-6" y="${y}" text-anchor="end" class="axis mono">${esc(epochToHHMM(t0))}</text>`; } // Colorbar const cbH = gridH; const cbX = gridW + COLORBAR_PAD; const cbY = 0; const cbSteps = Math.min(60, Math.max(24, Math.floor(cbH / 6))); // scale with height let cbRects = ""; for (let i = 0; i < cbSteps; i++) { const tt = i / (cbSteps - 1); const v = minV + tt * (maxV - minV); const y = cbY + (1 - tt) * cbH; cbRects += `<rect x="${cbX}" y="${y.toFixed(2)}" width="${COLORBAR_W}" height="${(cbH / cbSteps).toFixed( 2 )}" fill="${colorSequential(v, minV, maxV)}" />`; } const svgW = gridW + COLORBAR_PAD + COLORBAR_W + 54; const svgH = gridH + 22; const fMin = freq[0]; const fMax = freq[bins - 1]; // Clamp rendering width with scroll container const renderW = Math.min(svgW + 60, PANEL_MAX_W); return ` <div class="panel"> <div class="panel-head"> <div> <div class="panel-title">Channel ${esc(channelId)} · RxMER Heatmap</div> <div class="meta mono"> bins=${esc(String(bins))} · rows=${esc(String(T))} · cell=${esc(String(CELL_PX))}px · f=${esc(hzToMHz(fMin))}..${esc(hzToMHz(fMax))} MHz · rxmer=${esc(minV.toFixed(2))}..${esc(maxV.toFixed(2))} dB </div> </div> </div> <div class="panel-body"> <div class="scroll" style="max-width:${renderW}px;"> <svg viewBox="0 0 ${svgW} ${svgH}" class="hm-svg" role="img" aria-label="RxMER heatmap dense"> <g transform="translate(54, 6)"> ${rects} ${yLabels} ${xLabels} ${cbRects} <text x="${cbX + COLORBAR_W + 8}" y="10" class="axis mono">${esc(maxV.toFixed(1))}</text> <text x="${cbX + COLORBAR_W + 8}" y="${(cbH + 2).toFixed(2)}" class="axis mono">${esc(minV.toFixed(1))}</text> <text x="${cbX - 2}" y="${(cbH + 18).toFixed(2)}" text-anchor="end" class="axis mono">dB</text> </g> </svg> </div> <div class="meta"> Dense render (no hover, no borders). grid-template-columns: 1fr; gap: 14px; } .panel { background: linear-gradient(180deg, var(--panel), var(--panel2)); border: 1px solid var(--border); border-radius: 12px; box-shadow: 0 10px 24px var(--shadow); overflow: hidden; } .panel-head { padding: 12px 14px; border-bottom: 1px solid var(--border); display: flex; flex-wrap: wrap; gap: 10px 14px; align-items: baseline; justify-content: space-between; } .panel-title { font-size: 14px; font-weight: 700; } .panel-body { padding: 12px 14px 14px 14px; display: grid; gap: 10px; } .scroll { overflow-x: auto; padding-bottom: 6px; } .hm-svg { width: auto; /* allow natural width */ height: auto;
