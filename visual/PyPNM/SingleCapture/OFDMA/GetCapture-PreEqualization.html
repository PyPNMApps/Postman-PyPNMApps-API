`; function constructVisualizerPayload() { const resp = pm.response.json(); const analysis = (resp && resp.data && Array.isArray(resp.data.analysis)) ? resp.data.analysis : []; if (!analysis.length) return { channels: [] }; const channelMap = new Map(); function toFixedOrNA(v, digits) { if (typeof v !== 'number' || !isFinite(v)) return 'N/A'; return v.toFixed(digits); } function sanitizeMac(value, fallback) { var fb = (fallback === undefined || fallback === null) ? 'N/A' : fallback; if (value === undefined || value === null) return fb; var text = String(value).trim(); if (!text) return fb; var compact = text.replace(/[^0-9a-f]/gi, '').toLowerCase(); if (compact.length !== 12) return text; if (text.indexOf(':') !== -1) return compact.match(/.{1,2}/g).join(':'); if (text.indexOf('-') !== -1) return compact.match(/.{1,2}/g).join('-'); if (text.indexOf('_') !== -1) return compact.match(/.{1,2}/g).join('_'); return compact; } analysis.forEach(item => { const channelId = item.channel_id; if (channelId === undefined || channelId === null) return; if (!channelMap.has(channelId)) { channelMap.set(channelId, { channelId: channelId, macAddress: sanitizeMac(item.mac_address, 'N/A'), subcarrierZeroFreq: item.subcarrier_zero_frequency ?? 'N/A', series: [] }); } const ver = item.pnm_header && item.pnm_header.file_type_version !== undefined ? item.pnm_header.file_type_version : 'N/A'; const freqHz = item.carrier_values && Array.isArray(item.carrier_values.frequency) ? item.carrier_values.frequency : []; const frequencies = freqHz.map(f => { if (typeof f !== 'number' || !isFinite(f)) return 'N/A'; return (f / 1000000).toFixed(3); }); const magnitudes = item.carrier_values && Array.isArray(item.carrier_values.magnitudes) ? item.carrier_values.magnitudes : []; const groupDelays = item.carrier_values && item.carrier_values.group_delay && Array.isArray(item.carrier_values.group_delay.magnitude) ? item.carrier_values.group_delay.magnitude : []; const stats = item.signal_statistics || {}; const echoesRaw = item.echo && item.echo.report && Array.isArray(item.echo.report.echoes) ? item.echo.report.echoes : []; const echoes = echoesRaw.map(echo => ({ binIndex: echo.bin_index, time: (typeof echo.time_s === 'number' ? (echo.time_s * 1000000).toFixed(4) : 'N/A'), amplitude: (typeof echo.amplitude === 'number' ? echo.amplitude.toFixed(6) : 'N/A'), distanceM: (typeof echo.distance_m === 'number' ? echo.distance_m.toFixed(2) : 'N/A'), distanceFt: (typeof echo.distance_ft === 'number' ? echo.distance_ft.toFixed(2) : 'N/A') })); channelMap.get(channelId).series.push({ version: ver, frequencies: frequencies, magnitudes: magnitudes, groupDelays: groupDelays, stats: { mean: toFixedOrNA(stats.mean, 3), median: toFixedOrNA(stats.median, 3), std: toFixedOrNA(stats.std, 3), power: toFixedOrNA(stats.power, 3), peakToPeak: toFixedOrNA(stats.peak_to_peak, 3), crestFactor: toFixedOrNA(stats.crest_factor, 3) }, echoes: echoes }); }); const channels = Array.from(channelMap.values()).map(ch => { ch.series.sort((a, b) => String(a.version).localeCompare(String(b.version))); return ch; }); channels.sort((a, b) => Number(a.channelId) - Number(b.channelId)); return { channels }; } pm.visualizer.set(template, constructVisualizerPayload());
