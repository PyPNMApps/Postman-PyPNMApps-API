String(sys.MODEL).trim() : 'N/A' }; } function formatCaptureTime(raw) { if (raw === undefined || raw === null || raw === '') return null; if (typeof raw === 'number' && isFinite(raw)) { var ms = raw > 1e12 ? raw : raw * 1000; var d = new Date(ms); if (isNaN(d.getTime())) return null; return d.toISOString().slice(0, 19).replace('T', ' ') + ' UTC'; } var n = Number(raw); if (!isNaN(n) && isFinite(n)) return formatCaptureTime(n); var d2 = new Date(raw); if (isNaN(d2.getTime())) return String(raw); return d2.toISOString().slice(0, 19).replace('T', ' ') + ' UTC'; } function constructVisualizerPayload() { var response = pm.response.json(); var data = (response && response.data) ? response.data : {}; var analysis = (data.analysis && data.analysis[0]) ? data.analysis[0] : {}; var prim = (data.primative && data.primative[0]) ? data.primative[0] : {}; var signalAnalysis = (analysis && analysis.signal_analysis) ? analysis.signal_analysis : {}; var windowAverage = (signalAnalysis && signalAnalysis.window_average) ? signalAnalysis.window_average : {}; var sys = (((analysis || {}).device_details || {}).system_description) || {}; var pnmHeader = ((analysis || {}).pnm_header) || ((prim || {}).pnm_header) || null; var freqs = Array.isArray(signalAnalysis.frequencies) ? signalAnalysis.frequencies : []; return { deviceInfo: (function () { var s = sanitizeSystemDescription(sys); return { macAddress: sanitizeMac(prim.mac_address || response.mac_address, 'N/A'), MODEL: s.MODEL, VENDOR: s.VENDOR, SW_REV: s.SW_REV, HW_REV: s.HW_REV, BOOTR: s.BOOTR }; })(), captureTime: formatCaptureTime((pnmHeader || {}).capture_time), frequencyLabelsMHz: freqs.map(function (f) { return (Number(f) / 1000000).toFixed(0); }), magnitudes: Array.isArray(signalAnalysis.magnitudes) ? signalAnalysis.magnitudes : [], movingAverageMagnitudes: Array.isArray(windowAverage.magnitudes) ? windowAverage.magnitudes : [], binCount: freqs.length, binBandwidthMHz: signalAnalysis.bin_bandwidth ?
